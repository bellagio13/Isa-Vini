<script>
  // Função contador regressivo desde 28/07/2019 até agora
  function atualizarContador() {
    const inicioRelacionamento = new Date('2019-07-28T00:00:00');
    const agora = new Date();
    let diff = agora - inicioRelacionamento;

    const msPorSegundo = 1000;
    const msPorMinuto = msPorSegundo * 60;
    const msPorHora = msPorMinuto * 60;
    const msPorDia = msPorHora * 24;
    const msPorMes = msPorDia * 30.44; // média
    const msPorAno = msPorDia * 365.25;

    const anos = Math.floor(diff / msPorAno);
    diff -= anos * msPorAno;

    const meses = Math.floor(diff / msPorMes);
    diff -= meses * msPorMes;

    const dias = Math.floor(diff / msPorDia);
    diff -= dias * msPorDia;

    const horas = Math.floor(diff / msPorHora);
    diff -= horas * msPorHora;

    const minutos = Math.floor(diff / msPorMinuto);
    diff -= minutos * msPorMinuto;

    const segundos = Math.floor(diff / msPorSegundo);

    const texto = `${anos} ano${anos !== 1 ? 's' : ''}, ${meses} mês${meses !== 1 ? 'es' : ''}, ${dias} dia${dias !== 1 ? 's' : ''}, ${horas} hora${horas !== 1 ? 's' : ''}, ${minutos} minuto${minutos !== 1 ? 's' : ''} e ${segundos} segundo${segundos !== 1 ? 's' : ''}`;

    document.getElementById('contador').textContent = `Estamos juntos há ${texto}`;
  }

  // Atualiza o contador a cada segundo
  setInterval(atualizarContador, 1000);
  atualizarContador();

  // Lógica dos botões
  const btnSim = document.getElementById('btnSim');
  const btnNao = document.getElementById('btnNao');
  const telaInicio = document.getElementById('inicio');
  const telaGaleria = document.getElementById('galeria');

  // Para movimentar o botão "Não" que foge
  btnNao.addEventListener('mouseenter', () => {
    const maxX = window.innerWidth - btnNao.offsetWidth;
    const maxY = window.innerHeight - btnNao.offsetHeight;

    // Posiciona em local aleatório dentro da janela
    const novoX = Math.floor(Math.random() * maxX);
    const novoY = Math.floor(Math.random() * maxY);

    btnNao.style.position = 'absolute';
    btnNao.style.left = `${novoX}px`;
    btnNao.style.top = `${novoY}px`;
  });

  btnSim.addEventListener('click', () => {
    telaInicio.style.display = 'none';
    telaGaleria.style.display = 'flex';
    tocarMusica();
  });

  // Função para tocar música via API do YouTube
  function tocarMusica() {
    // Criar o player do YouTube via API JS para permitir autoplay e controle
    if (!window.YT) {
      // Carregar a API do YouTube
      var tag = document.createElement('script');
      tag.src = "https://www.youtube.com/iframe_api";
      document.body.appendChild(tag);
    } else {
      criarPlayer();
    }
  }

  // Função que cria player e inicia a música
  function criarPlayer() {
    window.player = new YT.Player('youtube-player', {
      events: {
        'onReady': function(event) {
          event.target.setVolume(50);
          event.target.playVideo();
        }
      }
    });
  }

  // Quando API carregada, cria player
  function onYouTubeIframeAPIReady() {
    criarPlayer();
  }
</script>
